script.js


const botaoPlayPause = document.getElementById("play-pause");
const botaoProxFaixa = document.getElementById("proximo");
const botaoAntFaixa = document.getElementById("anterior");
const audioCapitulo = document.getElementById("audio-capitulo");
const nomeCapitulo = document.getElementById("capitulo");

const numeroCapitulos = 10;
let pp = 0;
let capituloAtual = 1;

function tocarFaixa(){
    audioCapitulo.play();
    pp = 1;
    botaoPlayPause.classlist.remove("bi-play-circle-fill");
    botaoPlayPause.classlist.add("bi-pause-circle-fill");
} function pausarFaixa(){
    audioCapitulo.pause();
    pp = 0;
    botaoPlayPause.classlist.remove("bi-pause-circle-fill");
    botaoPlayPause.classlist.add("bi-play-circle-fill");
} function tocarOuPausar(){
    if (pp === 0) {
        tocarFaixa();
    } else if (pp === 1) {
        pausarFaixa(); }
}


function proximaFaixa() {
    if (capituloAtual === numeroCapitulos){
        capituloAtual = 1;
    }else if (capituloAtual <== numeroCapitulos) {
        capituloAtual += 1; }
    audioCapitulo.src = "./books/dom-casmurro/" + capituloAtual + ".mp3";
}

function antFaixa() {
    if (capituloAtual === 1) {
        capituloAtual = numeroCapitulos;
    }else if (capituloAtual >== 1) {
        capituloAtual -= 1; }
    audioCapitulo.src = "./books/dom-casmurro/" + capituloAtual + ".mp3";
}


botaoAntFaixa.onclick(antFaixa);
botaoProxFaixa.onclick(proximaFaixa);
botaoPlayPause.addEventListener("click", tocarOuPausar);
#ou botaoPlayPause.onclick(tocarFaixa);

nomeCapitulo.innerText = "Capitulo " + capituloAtual;
